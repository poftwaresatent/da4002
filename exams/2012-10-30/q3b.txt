int recurse (char *str, char **arr, int aa, int bb)
{
  int cc, dd;
  if (aa > bb) {
    return -1;
  }
  cc = (aa + bb) / 2;
  dd = slow_e (str, arr[cc]);
  if (dd > 0) {
    return recurse (str, arr, cc + 1, bb);
  }
  if (dd < 0) {
    return recurse (str, arr, aa, cc - 1);
  }
  return cc;
}

int function_two (char *str, char **arr, int nn)
{
  return recurse (str, arr, 0, nn - 1);
}
