\begin{align*}
  T_\text{VPM}(N) &= t_1 \left( \frac{N(N-1) - 2}{4} + 1 \right) \\
  &\text{with } \; t_1 = 0.4s \\
  T_\text{VPM} &\in O(N^2) \\
  T_\text{CTO}(N) &= t_2\frac{N}{5} + t_3N\log(N/5) \\
  &\text{with } \; t_2 = 5s \; \text{and } t_3 = 0.5s \\
  T_\text{CTO} &\in O(N\log N)
\end{align*}
